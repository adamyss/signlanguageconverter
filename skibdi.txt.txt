#define YELLOW   0xFFE0

Adafruit_NeoMatrix matrix = Adafruit_NeoMatrix(
  32, 8, LED_PIN,
  NEO_MATRIX_TOP + NEO_MATRIX_LEFT +
  NEO_MATRIX_COLUMNS + NEO_MATRIX_ZIGZAG,
  NEO_GRB + NEO_KHZ800
);

   pinMode(BUTTON_PIN, INPUT);

 matrix.fillScreen(WHITE);

void loop() {
 if (digitalRead(BUTTON_PIN) == HIGH && canCheckAgain == true) {
recording = !recording;
Serial.print("HIT BUTTON");
canCheckAgain = false;
if(recording == false){
  Serial.print("CHECKING");
  checkSymbol();
  delay(5000);
}
 } else if (digitalRead(BUTTON_PIN) == LOW){
  canCheckAgain = true;
 }
sensor = analogRead(A0);
sensor3 = analogRead(A2);
sensor4 = analogRead(A4);
sensor5 = analogRead(A5);
Serial.println(sensor5);
sensor2 = analogRead(A1);
 matrix.setCursor(0, 0);
 matrix.fillScreen(WHITE);
   matrix.print('\n');
   matrix.setTextColor(BLACK);
  if(recording){
    matrix.print("RECORDING");
  }
 if(reset == true){
        reset = false;
        matrix.begin();
        Serial.print("no Screen\n");
      }
     if(sensor > 25 && sensor2 > 25){
       matrix.print("HOORAY");
      } else if (sensor > 25 && sensor2 < 25){
        matrix.print("WILL");
         
      } else if(sensor2 > 25 & sensor < 25){
          matrix.print("MID");
